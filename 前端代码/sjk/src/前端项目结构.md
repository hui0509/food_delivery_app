# 前端项目结构方案



```
src/
├── api/                    # API接口管理
│   └── index.js           # API配置
├── assets/                 # 静态资源
│   └── logo.png           # 应用图标
├── components/             # 通用组件
│   ├── common/            # 通用组件（如按钮、表单等）
│   ├── layout/            # 布局组件（如头部、侧边栏等）
│   └── ui/                # UI基础组件
├── views/                  # 页面组件（按业务模块组织）
│   ├── auth/              # 认证相关页面
│   │   ├── Login.vue      # 登录页面（原MyLogReg.vue）
│   │   └── Register.vue   # 注册页面
│   ├── user/              # 用户相关页面
│   │   ├── Profile.vue    # 用户信息（原MyUser.vue）
│   │   ├── orders/        # 用户订单管理
│   │   │   ├── UserFinished.vue
│   │   │   ├── UserSending.vue
│   │   │   ├── UserUnsend.vue
│   │   │   └── OrderIssues.vue
│   │   └── profile/       # 用户信息管理
│   │       ├── AddressManagement.vue
│   │       ├── ChangePassword.vue
│   │       └── PersonalInfo.vue
│   ├── admin/             # 管理员页面
│   │   ├── Dashboard.vue  # 管理员仪表板（原MyManage.vue）
│   │   ├── UserManagement.vue
│   │   ├── ShopManagement.vue
│   │   ├── DispatcherManagement.vue
│   │   ├── orders/        # 管理员订单管理
│   │   │   ├── BeSended.vue
│   │   │   ├── BeSending.vue
│   │   │   ├── ManageOrderIssues.vue
│   │   │   └── UnSend.vue
│   │   └── delivery/      # 配送管理
│   │       ├── DeliveryEnded.vue
│   │       └── DeliveryUnended.vue
│   ├── store/             # 商家页面
│   │   ├── StoreAnalysis.vue
│   │   ├── StoreManage.vue
│   │   ├── StoreOrder.vue
│   │   ├── StoreShop.vue
│   │   └── dialogs/       # 商家对话框
│   │       ├── DishDialog.vue
│   │       ├── ImagePreview.vue
│   │       ├── ProfileDialog.vue
│   │       └── StoreDialog.vue
│   └── public/            # 公共页面
│       ├── Cart.vue       # 购物车
│       ├── Recommend.vue  # 推荐页面
│       ├── ShopDetail.vue # 店铺详情
│       └── ShopList.vue   # 店铺列表
├── router/                 # 路由配置
│   └── index.js           # 路由配置
├── utils/                  # 工具函数
├── styles/                 # 样式文件
├── config/                 # 配置文件
└── App.vue                 # 根组件
```
